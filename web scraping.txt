#the beautiful soup constructor allows us to parse a document
soup = BeautifulSoup(html, 'html5lib') #represents the document as a nested data structure (a set of tree like objects)

tag_object = soup.h3

find_all: a filter, lets you filter based on a tags name/text etc

from bs4 import BeautifulSoup

html = "<table><tr><th>Name</th><th>Age</th></tr>...</table>"
table = BeautifulSoup(html, 'html5lib')
table_rows=table.find_all(name='tr')

table_rows: looks through all the descendants and finds the descendants that match your filters
gives you a python iterable (like a list), with each element as a tag object for tr
first_row = table_rows[0]
will extract <tr><th>Name</th><th>Age</th></tr>

first_row.th will extract <th>Name</th> and <th>Age</th>
can also use first_row.td for the remaining rows
for i,row in enumerate(table_rows):
    print("row", i)
    cells=row.find_all("td")

    for j,cell in enumerate(cells):
        print("column", j, "cell", cell)  #can replace cell with cell.text.strip() to just get the table's cell values
